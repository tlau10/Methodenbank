179 bool __fastcall runbatfile(char* file, ThrowErrorMsg* err) {
180  STARTUPINFO stup;
181  PROCESS_INFORMATION pinf;
182  memset(&stup, 0, sizeof(STARTUPINFO));
183  stup.wShowWindow = SW_HIDE;
184  stup.dwFlags = STARTF_USESHOWWINDOW;
185  stup.cb = sizeof(STARTUPINFO);
186  char cmd[1000];
187  char* comspec = getenv("COMSPEC");
188  if (comspec) strcpy(cmd, comspec);
189  else {
190  comspec = "C:\\WINNT\\system32\\CMD.EXE";
191  if (!FileExists(comspec)) comspec = "C:\\WINDOWS\\system32\\CMD.EXE";
192  if (!FileExists(comspec)) comspec = "C:\\WINDOWS\\COMMAND.COM";
193  }
194  if (!FileExists(comspec)) return false;
195  char batfile[1000];
196  strcpy(batfile, "/C ");
197  strcat(batfile, file);
198  if (!CreateProcess(comspec, batfile, NULL, NULL, FALSE,
199  NORMAL_PRIORITY_CLASS, NULL, reg_getworkdir(), &stup,&pinf)) return false;
200  HANDLE ProcessHandle = OpenProcess(SYNCHRONIZE, false, pinf.dwProcessId);
201  int rval = WaitForSingleObject(ProcessHandle, 5000);
202  if (rval == WAIT_TIMEOUT) {
203  char msg[1000];
204  sprintf(msg, "5000 ms Execution-Timeout: CMD.EXE %s", batfile);
205  err->PutError(msg);
206  }
207  return true;
208 }
