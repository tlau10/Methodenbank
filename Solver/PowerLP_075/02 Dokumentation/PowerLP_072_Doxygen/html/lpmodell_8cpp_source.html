267 void __fastcall LPModell::ResToBounds() {
268  // Restriktionen durchlaufen, um zu prüfen ob eine Restriktion existiert die
269  // nur 1 Variable enthält, bsp X2 < 3 oder X4 > 4, d.h.
270  // die Anzahl der Variablen = 1 UND die einzige Variable ist mit 1 gewichtet
271  // Dann wird die Restriktion in eine Bounds-Angabe umgewandelt
272  // Funktion wird benötigt für Dual-Umwandlung
273  for (int y = 0; y<cress; y++) {
274  int appearance = 0;
275  int var = 0;
276  for (int x = 0; x < cvars ;x++) {
277  if (getRes(x,y) != 0.0) {
278  appearance ++;
279  var = x;
280  }
281  }
282  float resvalue = getRes(var, y);
283  if (appearance == 1 && (resvalue == 1.0 || resvalue == -1.0)) {
284  if (resvalue == -1.0) { // Restriktion mit -1 durchmultiplizieren
285  if (getRel(y) == LTEQUAL) setRel(y, STEQUAL);
286  if (getRel(y) == STEQUAL) setRel(y, LTEQUAL);
287  setBV(y, -1.0*getBV(y));
288  }
289  switch (getRel(y)) {
290  case LTEQUAL:
291  if (getLBound(var) == 0 /*&& getLBound(var) <= getBV(y)*/) {
292  setBounds(var, getBV(y), getHBound(var));
293  delcs(y);
294  y--;
295  }
296  break;
297  case STEQUAL:
298  if (getHBound(var) == 0 /*|| getHBound(var) >= getBV(y)*/) {
299  setBounds(var, getLBound(var), getBV(y));
300  delcs(y);
301  y--;
302  }
303  break;
304  }
305  }
306  }
307 }

310 void __fastcall LPModell::RemoveDuplicateRes() {
311  bool redcount = 0;
312  for (int y = 0; y<cress-1; y++) {
313  for (int y2 = y+1; y2 < cress; y2++) {
314  bool equal = true;
315  bool equalcond = true;
316  for (int x = 0; x < cvars ;x++) {
317  if (getRes(x,y) != getRes(x,y2)) { equal = false; equalcond = false; break; }
318  }
319  if (getBV(y) != getBV(y2)) { equal = false; equalcond = false; break; }
320  if (getRel(y) != getRel(y2)) { equal = false; }
321  if (!((getRel(y) == LTEQUAL && getRel(y2) == STEQUAL) ||
322  (getRel(y) == STEQUAL && getRel(y2) == LTEQUAL))) {
323  equalcond = false;
324  }
325  if (equal) {
326  delcs(y);
327  y--;
328  redcount++;
329  break;
330  }
331  if (equalcond) {
332  // Gleichungen die sich nur in der Bedingung mit <= und >= unterscheiden
333  // entsprechen einer einzigen = Restriktion
334  setRel(y2, EQUAL);
335  delcs(y);
336  y--;
337  break;
338  }
339  }
340  }
341  if (redcount) {
342  char msg[1000];
343  sprintf(msg, "Es wurde(n) %i redundante Restriktion(en) gelöscht.",redcount);
344  PutError(msg);
345  }

351 void __fastcall LPModell::getDual(LPModell* Dual) {
352  Dual->setMinimize(!this->getMinimize()); // true or false
353  float mf = -1.0; // Multi - Faktor
354  if (this->getMinimize()) mf = 1.0; // wenn Dual = Minimalisierungs-Problem
355  Dual->setSize(cress, cvars); /* vertauscht !! DualSize(Restriktionen, Variablen)
356  PrimalSize(Variablen, Restriktionen) */
357  int nr = 0; // dual var
358  for (int x = 0; x < cress; x++) { // Restriktionen durchlaufen
359  Dual->setSize(nr+1, cvars);
360  switch (this->getRel(x)) { // Relationen der Restriktionen holen
361  case EQUAL:
362  // Werte im 2-dim Array gerade vertauschen
363  for (int y = 0; y < cvars; y++) Dual->setRes(nr, y, -mf*this->getRes(y,x));
364  Dual->setZF(nr, -mf*this->getBV(x)); // b-Werte Primal => ZF-Werte, 'nur' eindimensional
365  nr++; Dual->setSize(nr+1, cvars);
366  for (int y = 0; y < cvars; y++) Dual->setRes(nr, y, mf*this->getRes(y,x));
367  Dual->setZF(nr, mf*this->getBV(x));
368  break;
369  case LTEQUAL:
370  case LARGER:
371  for (int y = 0; y < cvars; y++) Dual->setRes(nr, y, mf*this->getRes(y,x));
372  Dual->setZF(nr, mf*this->getBV(x));
373  break;
374  case STEQUAL:
375  case SMALLER:
376  for (int y = 0; y < cvars; y++) Dual->setRes(nr, y, -mf*this->getRes(y,x));
377  Dual->setZF(nr, -mf*this->getBV(x));
378  break;
379  }
380  nr++;
381  }
382 
383  for (int x = 0; x < cvars; x++) { // Variablen durchlaufen (Primal)
384  // für Dual die Restriktionen
385  float bv = this->getZF(x);
386  // comfort: mit minus 1 multiplizieren falls B-Vector < 0
387  if (bv < 0) mf = -1;
388  else mf = 1;
389  Dual->setBV(x, mf*this->getZF(x)); // b-werte Dual sind ZF-Wert (Primal)
390  for (int y = 0; y < cress; y++) {
391  Dual->setRes(y,x, mf*Dual->getRes(y,x));
392  }
393  int rel = mf;
394  if (this->getMinimize()) rel *= -1;
395  if (rel < 0) Dual->setRel(x, STEQUAL); // smaller
396  else Dual->setRel(x, LTEQUAL); // larger
397 
398  }
399  Dual->RemoveDuplicateRes();
400 }

439 void __fastcall LPModell::splitIntN() { // fuer XA
440  int ccvars = cvars; // Anzahl Variablen
441  for (int x = 0; x < ccvars; x++) { // Variablen durchlaufen
442  if (getInteger(x) && getHBound(x) != 1) {
443  // prüfe ganzzahlige Variablen deren obere Grenze > 1
444  int addvars = getHBound(x);
445  // zusätzliche Variablen
446  // Beispiel Variable X1 obere Grenze = 3, d.h. X1 wird aufgeteilt in 3
447  // Variablen, X1 = X1_1 + X1_2 + X1_3, wobei X1_1 bis X1_3 0,1-Variablen
448  // sind
449  bool lbound = false;
450  if (getLBound(x) > 0) { // Berücksichtigung der unteren Grenze
451  lbound = true;
452  addvars -= getLBound(x)-1; // addvars ist die Differenz zwischen
453  // oberer und unterer Grenze
454  // Die zusätzlichen Variablen reduzieren sich mit der unteren Grenze
455  // Beispiel von oben: Untere Grenze = 1, obere Grenze = 3
456  // damit Variable X1 = 1 + X1_2 + X1_3
457  }
458  if (addvars == 0) continue; // zusätzliche Variablen nicht notwendig
459  // Beispiel Untere / Obere Grenze 2 / 3
460  // neue Restriktion + N Variablen
461  int z = cvars;
462  // zusätzliche Variablen hinzufügen + auf 0 setzen sowie neue Nebenbedingung
463  // Beispiel:
464  // - 1 * X1 + X1_1 + X1_2 + X1_3 = 0
465  // bei Beispiel mit unterer Grenze
466  // - X1 + X1_1 + 2 * X1_2 = 0
467  // die untere Grenze wird auch in den Bounds angegeben ( X1 >= 2, X1 <= 3)
468  setSize(cvars+addvars, cress+1); // !!! cvars = cvars + addvars
469  // !!! cress = cress + 1
470  for (int y = 0; y < cvars; y++) setRes(y, cress-1, 0); // neue Restriktion
471  // alles loeschen
472  for (; z < cvars; z++) {
473  setRes(z, cress-1, 1.0); // auf Faktor eins in Nebenbedingung setzen
474  setZF(z, 0); // in ZF mit 0 gewichten
475  setInteger(z, true); // 0,1 Variablen
476  }
477  setRel(cress-1, EQUAL); // relation
478  setRes(x, cress-1, -1.0); // Ursprüngliche Variable mit -1 in neue Nebenbedingung
479  setBV(cress-1, 0);
480  if (lbound) setRes(z-1, cress-1, getLBound(x)); // wenn untere Grenze, dann
481  // anderer Multiplikator
482  setInteger(x, false); // ursprüngliche Variable ist jetzt nicht mehr
483  // ganzzahlig, da durch die neue Nebenbedingung die 0,N-Ganzzahligkeit
484  // gewährleistet ist
485  }
486  }
