HKEY openPowerLPReg()
                      {
  HKEY Result;
  RegOpenKeyEx(HKEY_LOCAL_MACHINE, "Software", NULL, KEY_ALL_ACCESS, &Result);
  RegCreateKeyEx(Result, "PowerLP", NULL, NULL, REG_OPTION_NON_VOLATILE ,
    KEY_ALL_ACCESS, NULL,  &Result, &disposition);
  return Result;
}
