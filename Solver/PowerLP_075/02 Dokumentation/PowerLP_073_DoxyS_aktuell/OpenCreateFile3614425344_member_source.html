int __fastcall OpenCreateFile(ThrowErrorMsg *emsg, char *filename)
                                                                   {
  int f = open(filename,O_RDWR | O_BINARY | O_CREAT | O_TRUNC, S_IREAD | S_IWRITE);
  if (f < 0) {
    char msg[1000];
    sprintf(msg, "Datei %s konnte nicht geöffnet/erstellt werden",filename);
    emsg->PutError(msg);
  }
  return f;
}
