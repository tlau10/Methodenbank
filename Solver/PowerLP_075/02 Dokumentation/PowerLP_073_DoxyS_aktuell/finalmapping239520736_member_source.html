void XAParser::finalmapping()
                            {
  printf("- final variable mapping:\n");
  for (int x = 0; x < vars; ) {
    int y = 0;
    for (; y < 15; y++) {
      char msg[10];
      sprintf(msg,"X%02i",x+1);
      printf("%05s", msg);
      x++;
      if (x >= vars) { y++; break;}
    }
    x-= y;
    printf("\n");
    for (int y = 0; y < 15; y++) {
      printf("%05s",variables[x]);
      x++;
      if (x > vars) break;
    }
    printf("\n");
    printf("\n");
  }
}
