HKEY openPowerLPRegUSER()
                          {
  HKEY Result;
  RegOpenKeyEx(HKEY_CURRENT_USER, "Software", NULL, KEY_ALL_ACCESS, &Result);
  RegCreateKeyEx(Result, "PowerLP", NULL, NULL, REG_OPTION_NON_VOLATILE ,
    KEY_ALL_ACCESS, NULL,  &Result, &disposition);
  return Result;
}
