char * movetonext(char *z, ptype p, int *len=NULL)
                                                    {
  bool firstchar = true;
  while (z[0]) {
    char* oldz = z;
    z = ignorechars(z);
    if (z!=oldz) return z;  // at next
    switch (p) {
      case ALPHA:
      if (!(z[0] >= 'A' && z[0] <= 'Z' ||
            z[0] >= 'a' && z[0] <= 'z' ||
            z[0] >= '0' && z[0] <= '9' || z[0] == '\''
            || (!firstchar && ( z[0] == '.' || z[0] == '-') )
            ))
            return z;
      break;
      case NUMERIC:
        if ((z[0] < '0' || z[0] > '9') && z[0] != '.') {
          if (!(firstchar && z[0] == '+' || z[0] == '-')) return z;
        }
      break;
      case SIGN:
        if (z[0] != '+' && z[0] != '-') return z;
      break;
      case OPERATOR:
        if (z[0] != '<' && z[0] != '>' && z[0] != '=') return z;
      break;
    }
    z++;
    if (len) ++*len;
    firstchar = false;
  }
  z += strlen(z);
  if (len) *len+= strlen(z);
  return z;
