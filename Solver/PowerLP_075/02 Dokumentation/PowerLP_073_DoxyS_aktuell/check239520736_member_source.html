int XAParser::check()
                    {
  // check for empty vars
  for (int x = 0; x < vars ;x++) {
    int appearance = 0;
    for (int y = 0; y<nbs; y++) {
      if (factornb[y][x] != 0.0) appearance ++;
    }
    if (!appearance) {
      char msg[1000];
      sprintf(msg,"Warnung: Variable %s ist ungültig / erscheint in keiner Nebenbedingung.",variables[x]);
      emsg->PutError(msg);
      delvar(x);
      x--;
    }
  }
  return 0;
}
