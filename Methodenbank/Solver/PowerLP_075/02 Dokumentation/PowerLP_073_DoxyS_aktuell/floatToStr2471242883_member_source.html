char * floatToStr(char *str, float value)
                                         {
  float absv = fabs(value);
  if (absv > 999999) sprintf(str, "%f", value);
  else if (absv > 0 && absv < 0.0001) sprintf(str, "%.20f", value);
  else sprintf(str, "%g", value);
//  sprintf(str, "%g", value);
  strreplace(str, ".", ",");
  return str;
}
