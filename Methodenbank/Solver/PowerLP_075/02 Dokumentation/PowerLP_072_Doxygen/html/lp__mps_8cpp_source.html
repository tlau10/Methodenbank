68 int MPSParser::scanRows() {
69  res = 0;
70  while (readLF(file, line, sizeof(line))) { // holt die nächste Zeile in line
71  // aus der Section 'Rows'
72  pline ++; // Zeilenzähler
73  if (checkBreakLine()) break; // break condition - sucht die Schlüsselwörter im file
74  // first char
75  char* z = line; // Zeigerarray
76  char cname[sizeof(line)]; // cname = der Name der Restriktion (ZF)
77  bool cfinished = false;
78  while (strlen(z) && !cfinished) {
79  bool objective = false; // Zielfunktions-Kennzeichen
80  z = ignorechars(z); // eliminiert alle Leerstellen vor dem nächsten Element
81  if (z[0] == '*') break; // comment
82  // check for rowtype
83  cname[0] = 0;
84  if (z[0] >= 'A' && z[0] <= 'Z' || // Buchstabe A-Z oder a-z
85  z[0] >= 'a' && z[0] <= 'z') {
86  if (z[0] != 'N' && z[0] != 'E' && z[0] != 'L' && z[0] != 'G')
87  return 1; // erster Buchstabe muss N, E, L oder G sein, ansonsten Fehler
88  /*
89  N - Zielfunktion
90  E - Nebenbedingung =
91  L - Nebenbedingung <=
92  G - Nebenbedingung >=
93  */
94  char cond = z[0]; // *** Typ der Restriktion ***
95  if (cond == 'N') objective = true; // Zielfunktion
96  z = movetonext(z, ALPHA); // geht zum nächsten Eintrag in der Zeile
97  // Rückgabewert ist die (gekürzte) Zeile
98  // -> der Name der Restriktion
99 // z = ignorechars(z);
100  if (parseStr(z,cname) != 1) return 2; // *** Namen der Restriktion ***
101  if (!objective) { // keine Zielfunktion
102  LPM->setSize(LPM->getVarCount(), res+1); // hier ist es eine Restriktion
103  LPM->setResName(res, cname);
104  relation rel = STEQUAL; // Vorgabewert
105  switch (cond) {
106  case 'L': rel = STEQUAL; break;
107  case 'E': rel = EQUAL; break;
108  case 'G': rel = LTEQUAL; break;
109  }
110  LPM->setRel(res, rel);
111  res++; // Restriktion
112  } else {
113  LPM->setZFName(cname); // Name der Zielfunktion-Referenz (1. Durchlauf)
114  // braucht man später wieder für scanColums
115  }
116  }
117  break;
118  }
119  }
120  return 0;
121 }

123 int MPSParser::scanColumns() {
124  bool integer = false; // default : keine Ganzzahligkeit
125  while (readLF(file, line, sizeof(line))) {
126  pline ++;
127  if (checkBreakLine()) break; // break condition
128  // first char
129  char* z = line;
130  char vname[sizeof(line)]; // VariablenBezeichner
131  char cname[sizeof(line)];
132  int varindex = -1;
133  while (strlen(z)) {
134  z = ignorechars(z);
135  if (z[0] == '*') break; // comment
136  // check for varname
137  vname[0] = 0;
138  if (z[0] >= 'A' && z[0] <= 'Z' || // erste Element Anfang Buchstabe
139  z[0] >= 'a' && z[0] <= 'z') {
140  if (parseStr(z,vname) != 1) return 11; // *** VariablenName ***
141  z = movetonext(z, ALPHA); // nächster Eintrag
142 // varindex = getVarIndex(vname); // marker ident is not a var
143  } else return 11; // Fehler beim VariablenBez.
144  // check for resname
145  cname[0] = 0;
146  bool iszf = false;
147  int resindex = -1;
148  if (varindex >= 0 && LPM->getResNameIX(vname) >= 0) {
149  // zweite Spalte RESNAME WERT, dann ist vname = Restriktionsbezeichner
150  // nach Standard ist die Regel "Varname Resname Wert Resname Wert"
151  // es gibt aber auch files mit "Varname Resname Wert Varname Resname Wert"
152  if (LPM->isZFName(vname)) iszf = true; // es handelt sich um einen ZF-Wert
153  else resindex = LPM->getResNameIX(vname); // index Restriktionen
154  resindex = LPM->getResNameIX(vname);
155  // varindex vom letzten durchlauf
156  } else {
157  if (z[0] >= 'A' && z[0] <= 'Z' || // nächste Element Anfang Buchstabe
158  z[0] >= 'a' && z[0] <= 'z') {
159  if (parseStr(z,cname) != 1) return 12; // *** RestriktionenName ***
160  z = movetonext(z, ALPHA); // nächster Eintrag
161  if (LPM->isZFName(cname)) iszf = true; // es handelt sich um einen ZF-Wert
162  else resindex = LPM->getResNameIX(cname); // index Restriktionen
163  varindex = getVarIndex(vname); // index Variablen !!!
164  } else {
165  z = movetonext(z, ALPHA); // 'Marker' - wir einfach das
166  // naechste geholt
167  }
168  }
169  // check for value // *** Variablenwert ***
170  float value = 0.0;
171  // ist es eine Zahl ?
172  if (z[0] >= '0' && z[0] <= '9' || z[0] == '.' || z[0] == '-') {
173  if (parseFloat(z, &value) != 1) {return 13;}
174  z = movetonext(z, NUMERIC);
175  } else { // check for int marker
176  if (!parseStr(z, vname)) return 13; // in vname einkopieren !
177  if (resindex >= 0) return 13; // darf keinen RestriktionsIndex haben
178  if (!stricmp(vname, "'INTORG'")) { // the same as == 0
179  // Umschalter Ganzzahligkeit
180  integer = true;
181  break;
182  } else if (!stricmp(vname, "'INTEND'")) {
183  integer = false;
184  break;
185  } else return 16;
186  }
187  if (varindex < 0) return 14;
188  if (iszf) LPM->setZF(varindex, value); // setzen der Zielfunktionsvar.
189  else {
190  if (resindex < 0) return 15;
191  LPM->setRes(varindex, resindex, value); // setzen der Restriktionsvar.
192  // 2-dim Array
193  }
194  LPM->setInteger(varindex, integer); // ganzzahlig
195 // break; // next line
196  }
197  }
198  return 0; // alles gut gegangen !
199 }

313 int MPSParser::parse() {
314  int rval = 0;
315  bool rowsfound = false;
316  bool rhsfound = false; // b-wert
317  bool colsfound = false;
318  LPM->setSize(0,0);
319  while (readLF(file, line, sizeof(line))) { /* line hat das Format char[2048]
320  die while-Schleife liest das file zeilenweise
321  aus, bis ENDDATA, dann wird die Schleife mit Break
322  verlassen oder bread = 0 */
323  pline ++; // zaehlt die Zeilen mit
324  bool goback = true;
325  while (goback) {
326  goback = false;
327  if (strstr(line, "NAME")) { // Scans a string for the occurrence of a given substring
328  // in line steht immer eine Zeile des *.mps - files
329  LPM->setMinimize(false); // default
330  if ( strstr(line, "MINIMIZE") ) LPM->setMinimize(true); // Minimierung-Problem
331  else if (strstr(line, "MAXIMIZE")) LPM->setMinimize(false); // Maximierungs-Problem
332  else LPM->PutError("Richtung der Optimierung (Minimierung/Maximierung) in MPS-Datei nicht angegeben.");
333  }
334  if (strstr(line, "ROWS")) { // Scans a string for the occurrence of a given substring
335  rval = scanRows(); // macht alle Zeilen durch
336  goback = true;
337  rowsfound = true;
338  }
339  if (!stricmp(line, "COLUMNS")) { // the same as == 0
340  rval = scanColumns();
341  goback = true;
342  colsfound = true;
343  }
344  if (!stricmp(line, "RHS")) {
345  rval = scanRHS();
346  goback = true;
347  rhsfound = true;
348  }
349  if (!stricmp(line, "BOUNDS")) {
350  rval = scanBounds();
351  goback = true;
352  }
353  if (!stricmp(line, "ENDATA")) {
354  break;
355  }
356  if (!stricmp(line, "RANGES")) {
357  // scan Ranges
358  rval = scanRanges();
359  goback = true;
360  }
361  }
362  if (rval) { // rval = 0
363  char msg[1000];
364  sprintf(msg, "Parser-Fehler bei Linie %i: %s", pline, line);
365  LPM->PutError(msg);
366  return rval;
367  }
368  // ... naechste Zeile einlesen ...
369  }
370  if (!rowsfound || !rhsfound || !colsfound) {
371 // LPM->PutError("Fehler: Diese Datei ist keine MPS Datei.");
372  return 101;
373  }
374 // check();
375 // finalmapping();
376  return 0;
377 }

379 char* floatToStrENDOT(char* str, float value) {
380  float absv = fabs(value);
381  if (absv > 999999) sprintf(str, "%f", value);
382  else if (absv > 0 && absv < 0.0001) sprintf(str, "%.9f", value);
383  else sprintf(str, "%g", value);
384  if (!strchr(str,'.')) strcat(str, ".");
385  return str;

428 bool __fastcall PutLPtoMPSFile(LPModell* LPM, char* filename, bool boundssupport) {
429  int f = OpenCreateFile(LPM, filename);
430  if (f < 0) return false;
431  // Grenz-Bedingungen zaehlen fuer den Fall das boundssupport = false
432  int boundsres = 0;
433  writeStr(f, "NAME Power-LP "); // mps - Power-LP - Dateien
434  if (LPM->getMinimize()) writeLF(f, "MINIMIZE");
435  else writeLF(f, "MAXIMIZE");
436  int vars= LPM->getVarCount(); // Anzahl Variablen
437  int res = LPM->getResCount(); // Anzahl Restriktionen
438  char str[200];
439  char rs[30];
440  writeLF(f, "ROWS"); // --- ROWS - Section ---
441  writeLF(f, " N ZF"); // Zielfunktionszeile
442  for (int x = 0; x<res; x++) {
443  char* pstr = "";
444  switch (LPM->getRel(x)) { // eine Restriktion nach der anderen
445  case EQUAL: pstr = "E"; break;
446  case LTEQUAL: pstr = "G"; break;
447  case STEQUAL: pstr = "L"; break;
448  case LARGER: pstr = "G"; break;
449  case SMALLER: pstr = "L"; break;
450  }
451  writeStr(f," ");
452  writeStr(f,pstr);
453  writeStr(f," ");
454  sprintf(str, "R%i", x+1);
455  writeLF(f,str);
456  }
457  if (!boundssupport) {
458  for (int x = 0; x<vars; x++) {
459  float lb = LPM->getLBound(x); // lower bound
460  if (lb != 0) {
461  boundsres++;
462  sprintf(str, " G RB%i", boundsres);
463  writeLF(f,str); // schreiben als Restriktion
464  }
465  float hb = LPM->getHBound(x); // higher bound
466  if (hb != 0) {
467  boundsres++;
468  sprintf(str, " L RB%i", boundsres);
469  writeLF(f,str); // schreiben als Restriktionen
470  }
471  }
472  }
473  writeLF(f, "COLUMNS"); // --- COLUMNS - Section ---
474  bool intmode = false;
475  int markerno = 0;
476  boundsres = 0;
477  for (int x = 0; x<vars; x++) { // geht alle Variablen des LP durch
478  float value = LPM->getZF(x); // *** Zielfunktion ***
479  char cvar[30];
480  sprintf(str,"X%i", x+1);
481  sprintf(cvar, "%-010s", str);
482  if (value != 0.0) { // *** Ganzzahligkeit ***
483  if (LPM->getInteger(x) != intmode) { // Ganzzahligkeit für diese Variable
484  sprintf(str, " MARK%04i 'MARKER' ", markerno);
485  markerno++;
486  writeStr(f,str);
487  if (!intmode) writeLF(f,"'INTORG'"); // Anfang
488  else writeLF(f,"'INTEND'"); // Ende
489  intmode = !intmode;
490  }
491  writeStr(f, " ");
492  writeStr(f,cvar);
493  writeStr(f,"ZF ");
494  floatToStrENDOT(str, value);
495  writeLF(f,str);
496  } // if (value != 0.0)
497  // für die Variable ...
498  for (int y = 0; y < res; y++) { // alle Restriktionen durchlaufen
499  float value = LPM->getRes(x,y); // 2-dimensionales Array
500  if (value != 0.0) {
501  if (LPM->getInteger(x) != intmode) {
502  sprintf(str, " MARK%04i 'MARKER' ", markerno);
503  markerno++;
504  writeStr(f,str);
505  if (!intmode) writeLF(f,"'INTORG'");
506  else writeLF(f,"'INTEND'");
507  intmode = !intmode;
508  }
509  writeStr(f," ");
510  writeStr(f,cvar);
511  sprintf(rs, "R%i",y+1);
512  sprintf(str, "%-010s", rs);
513  writeStr(f,str);
514  floatToStrENDOT(str, LPM->getRes(x,y));
515  writeLF(f,str);
516  }
517  }
518  // bounds // ... nach den Restriktionen
